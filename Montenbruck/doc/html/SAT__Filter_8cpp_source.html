59 void InvUpper(const Matrix& R, Matrix& T)
60 {
61 
62  const int N = R.size1(); // Dimension of R and T
63 
64  int i,j,k;
65  double Sum;
66 
67  if ( R.size2()!=N || T.size1()!=N || T.size2()!=N ) {
68  cerr << " ERROR: Incompatible shapes in InvUpper" << endl;
69  exit(1);
70  }
71 
72  // Check diagonal elements
73 
74  for (i=0; i<N; i++)
75  if ( R(i,i) == 0.0 ) {
76  cerr << " ERROR: Singular matrix in InvUpper" << endl;
77  exit(1);
78  }
79  else {
80  // Compute the inverse of i-th diagonal element.
81  T(i,i) = 1.0/R(i,i);
82  };
83 
84  // Calculate the inverse T = R^(-1)
85 
86  for (i=0; i<N-1; i++)
87  for (j=i+1; j<N; j++) {
88  Sum = 0.0; for (k=i; k<=j-1; k++) Sum += T(i,k)*R(k,j);
89  T(i,j) = -T(j,j)*Sum;
90  };

92 
93 }

134 void LSQ::Init(const Vector& x, // a priori parameters
135  const Matrix& P) // a priori covariance
136 {
137 
138  // Variables
139 
140  int i,j,k;
141  double Sum;
142 
143 
144  // Reset number of data equations
145 
146  n = 0;
147 
148  // Start the factorization of matrix P. Compute upper triangular
149  // factor R of P, where P = (R)*(R^T). Proceed backward column
150  // by column.
151 
152  for (j=N-1;j>=0;j--) {
153 
154  // Compute j-th diagonal element.
155 
156  Sum = 0.0;
157  for (k=j+1;k<=N-1;k++) Sum += R(j,k)*R(j,k);
158  R(j,j) = sqrt(P(j,j)-Sum);
159 
160  // Complete factorization of j-th column.
161 
162  for (i=j-1;i>=0;i--) {
163  Sum = 0.0;
164  for (k=j+1;k<=N-1;k++) Sum += R(i,k)*R(j,k);
165  R(i,j) = (P(i,j)-Sum)/R(j,j);
166  };
167 
168  };
169 
170  // Replace R by its inverse R^(-1)
171 
172  InvUpper(R,R);
173 
174  // Initialize right hand side
175 
176  d = R*x;

186 void LSQ::Accumulate (const Vector& A, double b, double sigma)
187 {
188  // Variables
189  int i,j;
190  double c,s,h;
191  Vector a(N);
192 
193  // Weighting
194  a = A/sigma; // Normalize A
195  b = b/sigma; // Normalize b
196 
197  // Construct and apply Givens plane rotation.
198  for (i=0; i<N; i++)
199  {
200  // Construct the rotation and apply it to
201  // eliminate the i-th element of a.
202  if ( R(i,i)==0.0 && a(i)==0.0 ) {
203  c = 1.0; s = 0.0; R(i,i) = 0.0;
204  }
205  else {
206  h = sqrt ( R(i,i)*R(i,i) + a(i)*a(i) );
207  if (R(i,i)<0.0) h=-h;
208  c = R(i,i)/h;
209  s = a(i)/h;
210  R(i,i) = h;
211  };
212 
213  a(i) = 0.0;
214 
215  // Apply the rotation to the remaining elements of a
216  for (j=i+1; j<N; j++) {
217  h = +c*R(i,j)+s*a(j);
218  a(j) = -s*R(i,j)+c*a(j);
219  R(i,j) = h;
220  }
221 
222  // Apply the rotation to the i-th element of d
223  h = +c*d(i)+s*b;
224  b = -s*d(i)+c*b;
225  d(i) = h;
226  }
227 
228  // Increment number of data equations
229  n++;
230 
231 }

238 void LSQ::Solve (Vector& x)
239 {
240  // Variables
241  int i,j; i=j=0;
242  double Sum=0.0;
243 
244  // Check for singular matrix
245  for (i=0;i<N;i++)
246  if ( R(i,i) == 0.0 ) {
247  cerr << " ERROR: Singular matrix R in LSQ::Solve()" << endl;
248  exit(1);
249  };
250 
251  // Solve Rx=d for x_n,...,x_1 by backsubstitution
252  x(N-1) = d(N-1) / R(N-1,N-1);
253  for (i=N-2;i>=0;i--) {
254  Sum = 0.0;
255  for (j=i+1;j<N;j++) Sum += R(i,j)*x(j);
256  x(i) = ( d(i) - Sum ) / R(i,i);
257  };
