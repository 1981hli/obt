183 //------------------------------------------------------------------------------
184 //
185 // Sun
186 //
187 // Purpose:
188 //
189 // Computes the Sun's geocentric position using a low precision
190 // analytical series
191 //
192 // Input/Output:
193 //
194 // Mjd_TT Terrestrial Time (Modified Julian Date)
195 // <return> Solar position vector [m] with respect to the
196 // mean equator and equinox of J2000 (EME2000, ICRF)
197 //
198 //------------------------------------------------------------------------------
199 
200 Vector Sun (double Mjd_TT)
201 {
202  // Constants
203 
204  const double eps = 23.43929111*Rad; // Obliquity of J2000 ecliptic
205  const double T = (Mjd_TT-MJD_J2000)/36525.0; // Julian cent. since J2000
206 
207  // Variables
208 
209  double L, M, r;
210  Vector r_Sun(3);
211 
212  // Mean anomaly, ecliptic longitude and radius
213 
214  M = pi2 * Frac ( 0.9931267 + 99.9973583*T); // [rad]
215  L = pi2 * Frac ( 0.7859444 + M/pi2 +
216  (6892.0*sin(M)+72.0*sin(2.0*M)) / 1296.0e3); // [rad]
217  r = 149.619e9 - 2.499e9*cos(M) - 0.021e9*cos(2*M); // [m]
218 
219  // Equatorial position vector
220 
221  r_Sun = R_x(-eps) * Vector(r*cos(L),r*sin(L),0.0);
222 
223  return r_Sun;
224

229 //
230 // Moon
231 //
232 // Purpose:
233 //
234 // Computes the Moon's geocentric position using a low precision
235 // analytical series
236 //
237 // Input/Output:
238 //
239 // Mjd_TT Terrestrial Time (Modified Julian Date)
240 // <return> Lunar position vector [m] with respect to the
241 // mean equator and equinox of J2000 (EME2000, ICRF)
242 //
243 //------------------------------------------------------------------------------
244 
245 Vector Moon (double Mjd_TT)
246 {
247  // Constants
248 
249  const double eps = 23.43929111*Rad; // Obliquity of J2000 ecliptic
250  const double T = (Mjd_TT-MJD_J2000)/36525.0; // Julian cent. since J2000
251 
252  // Variables
253 
254  double L_0, l,lp, F, D, dL, S, h, N;
255  double L, B, R, cosB;
256  Vector r_Moon(3);
257 
258 
259  // Mean elements of lunar orbit
260 
261  L_0 = Frac ( 0.606433 + 1336.851344*T ); // Mean longitude [rev]
262  // w.r.t. J2000 equinox
263  l = pi2*Frac ( 0.374897 + 1325.552410*T ); // Moon's mean anomaly [rad]
264  lp = pi2*Frac ( 0.993133 + 99.997361*T ); // Sun's mean anomaly [rad]
265  D = pi2*Frac ( 0.827361 + 1236.853086*T ); // Diff. long. Moon-Sun [rad]
266  F = pi2*Frac ( 0.259086 + 1342.227825*T ); // Argument of latitude
267 
268 
269  // Ecliptic longitude (w.r.t. equinox of J2000)
270 
271  dL = +22640*sin(l) - 4586*sin(l-2*D) + 2370*sin(2*D) + 769*sin(2*l)
272  -668*sin(lp) - 412*sin(2*F) - 212*sin(2*l-2*D) - 206*sin(l+lp-2*D)
273  +192*sin(l+2*D) - 165*sin(lp-2*D) - 125*sin(D) - 110*sin(l+lp)
274  +148*sin(l-lp) - 55*sin(2*F-2*D);
275 
276  L = pi2 * Frac( L_0 + dL/1296.0e3 ); // [rad]
277 
278  // Ecliptic latitude
279 
280  S = F + (dL+412*sin(2*F)+541*sin(lp)) / Arcs;
281  h = F-2*D;
282  N = -526*sin(h) + 44*sin(l+h) - 31*sin(-l+h) - 23*sin(lp+h)
283  +11*sin(-lp+h) - 25*sin(-2*l+F) + 21*sin(-l+F);

293  -171e3*cos(l+2*D) - 152e3*cos(l+lp-2*D);
294 
295  // Equatorial coordinates
296 
297  r_Moon = R_x(-eps) * Vector ( R*cos(L)*cosB, R*sin(L)*cosB, R*sin(B) );
298 
299  return r_Moon;

333 //------------------------------------------------------------------------------
334 //
335 // AccelHarmonic
336 //
337 // Purpose:
338 //
339 // Computes the acceleration due to the harmonic gravity field of the
340 // central body
341 //
342 // Input/Output:
343 //
344 // r Satellite position vector in the inertial system
345 // E Transformation matrix to body-fixed system
346 // GM Gravitational coefficient
347 // R_ref Reference radius
348 // CS Spherical harmonics coefficients (un-normalized)
349 // n_max Maximum degree
350 // m_max Maximum order (m_max<=n_max; m_max=0 for zonals, only)
351 // <return> Acceleration (a=d^2r/dt^2)
352 //
353 //------------------------------------------------------------------------------
354 
355 Vector AccelHarmonic (const Vector& r, const Matrix& E,
356  double GM, double R_ref, const Matrix& CS,
357  int n_max, int m_max )
358 {
359 
360  // Local variables
361 
362  int n,m; // Loop counters
363  double r_sqr, rho, Fac; // Auxiliary quantities
364  double x0,y0,z0; // Normalized coordinates
365  double ax,ay,az; // Acceleration vector
366  double C,S; // Gravitational coefficients
367  Vector r_bf(3); // Body-fixed position
368  Vector a_bf(3); // Body-fixed acceleration
369 
370  Matrix V(n_max+2,n_max+2); // Harmonic functions
371  Matrix W(n_max+2,n_max+2); // work array (0..n_max+1,0..n_max+1)
372 
373 
374  // Body-fixed position
375 
376  r_bf = E * r;
377 
378  // Auxiliary quantities
379 
380  r_sqr = Dot(r_bf,r_bf); // Square of distance
381  rho = R_ref*R_ref / r_sqr;
382 
383  x0 = R_ref * r_bf(0) / r_sqr; // Normalized
384  y0 = R_ref * r_bf(1) / r_sqr; // coordinates
385  z0 = R_ref * r_bf(2) / r_sqr;
386 
387  //
388  // Evaluate harmonic functions
389  // V_nm = (R_ref/r)^(n+1) * P_nm(sin(phi)) * cos(m*lambda)
390  // and
391  // W_nm = (R_ref/r)^(n+1) * P_nm(sin(phi)) * sin(m*lambda)
392  // up to degree and order n_max+1
393  //
394 
395  // Calculate zonal terms V(n,0); set W(n,0)=0.0
396 
397  V(0,0) = R_ref / sqrt(r_sqr);
398  W(0,0) = 0.0;
399 
400  V(1,0) = z0 * V(0,0);
401  W(1,0) = 0.0;
402 
403  for (n=2; n<=n_max+1; n++) {
404  V(n,0) = ( (2*n-1) * z0 * V(n-1,0) - (n-1) * rho * V(n-2,0) ) / n;
405  W(n,0) = 0.0;
406  };
407 
408  // Calculate tesseral and sectorial terms
409 
410  for (m=1; m<=m_max+1; m++) {
411 
412  // Calculate V(m,m) .. V(n_max+1,m)
413 
414  V(m,m) = (2*m-1) * ( x0*V(m-1,m-1) - y0*W(m-1,m-1) );

420  };

423  V(n,m) = ( (2*n-1)*z0*V(n-1,m) - (n+m-1)*rho*V(n-2,m) ) / (n-m);
424  W(n,m) = ( (2*n-1)*z0*W(n-1,m) - (n+m-1)*rho*W(n-2,m) ) / (n-m);

427  };

431  //
432

436  for (n=m; n<=n_max ; n++)

438  C = CS(n,0); // = C_n,0
439  ax -= C * V(n+1,1);
440  ay -= C * W(n+1,1);
441  az -= (n+1)*C * V(n+1,0);
442  }
443  else {
444  C = CS(n,m); // = C_n,m
445  S = CS(m-1,n); // = S_n,m
446  Fac = 0.5 * (n-m+1) * (n-m+2);
447  ax += + 0.5 * ( - C * V(n+1,m+1) - S * W(n+1,m+1) )
448  + Fac * ( + C * V(n+1,m-1) + S * W(n+1,m-1) );
449  ay += + 0.5 * ( - C * W(n+1,m+1) + S * V(n+1,m+1) )
450  + Fac * ( - C * W(n+1,m-1) + S * V(n+1,m-1) );
451  az += (n-m+1) * ( - C * V(n+1,m) - S * W(n+1,m) );
452  };
453 
454  // Body-fixed acceleration
455 
456  a_bf = (GM/(R_ref*R_ref)) * Vector(ax,ay,az);

459
