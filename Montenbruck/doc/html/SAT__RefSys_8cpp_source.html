649 Geodetic::Geodetic (Vector r, // Position vector [m]
650  double R_equ, // Equator radius [m]
651  double f) // Flattening
652 {
653 
654  const double eps = 1.0e3*eps_mach; // Convergence criterion
655  const double epsRequ = eps*R_equ;
656  const double e2 = f*(2.0-f); // Square of eccentricity
657 
658  const double X = r(0); // Cartesian coordinates
659  const double Y = r(1);
660  const double Z = r(2);
661  const double rho2 = X*X + Y*Y; // Square of distance from z-axis
662 
663  // Check validity of input data
664 
665  if (Norm(r)==0.0) {
666  cerr << " invalid input in Geodetic constructor" << endl;
667  lon=0.0; lat=0.0; h=-R_Earth;
668  return;
669  }
670 
671  // Iteration
672 
673  double dZ, dZ_new, SinPhi;
674  double ZdZ, Nh, N;
675 
676  dZ = e2*Z;
677  for(;;) {
678  ZdZ = Z + dZ;
679  Nh = sqrt ( rho2 + ZdZ*ZdZ );
680  SinPhi = ZdZ / Nh; // Sine of geodetic latitude
681  N = R_equ / sqrt(1.0-e2*SinPhi*SinPhi);
682  dZ_new = N*e2*SinPhi;
683  if ( fabs(dZ-dZ_new) < epsRequ ) break;
684  dZ = dZ_new;
685  }
686 
687  // Longitude, latitude, altitude
688 
689  lon = atan2 ( Y, X );
690  lat = atan2 ( ZdZ, sqrt(rho2) );
691  h = Nh - N;
692 
693 }

704  const double SinLat = sin(lat);
