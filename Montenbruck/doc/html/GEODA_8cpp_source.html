77 Matrix Phi(double n, double t) {
78 
79  const double c = cos(n*t);
80  const double s = sin(n*t);
81 
82  Matrix T(6,6);
83 
84  T(0,0) = 4-3*c; T(0,1) = 0; T(0,3) = s/n; T(0,4) = 2*(1-c)/n;
85  T(1,0) = 6*(s-n*t); T(1,1) = 1; T(1,3) = 2*(c-1)/n; T(1,4) = 4*s/n-3*t;
86  T(2,2) = c; T(2,5) = s/n;
87 
88  T(3,0) = 3*n*s; T(3,1) = 0; T(3,3) = c; T(3,4) = 2*s;
89  T(4,0) = 6*n*(c-1); T(4,1) = 0; T(4,3) = -2*s; T(4,4) = 4*c-3;
90  T(5,2) = -n*s; T(5,5) = c;
91 
92  return T;
93
