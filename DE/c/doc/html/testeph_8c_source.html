4 * program testeph *
5 * *
6 * *
7 * Testeph tests the jpl ephemeris reading and interpolating routine using *
8 * examples computed from the original ephemeris. *
9 * *
10 * Testeph contains the reading and interpolating subroutines that are of *
11 * eventual interest to the user. Once testeph is working correctly, the *
12 * user can extract those subroutines and the installation process is *
13 * complete. *
14 * *
15 * You must allow access to "testpo.xxx" to testeph. *
16 * "testpo.xxx" is the specially formatted text file that contains the test *
17 * cases for the ephemeris, dexxx. *
18 * *
19 * After the initial identifying text which is concluded by an "EOT" in *
20 * columns 1-3, the test file contains the following quantities: *
21 * *
22 * JPL ephemeris number *
23 * calendar date *
24 * julian ephemeris date *
25 * target number (1-mercury, ...,3-earth, ,,,9-pluto, 10-moon, 11-sun, *
26 * 12-solar system barycenter, 13-earth-moon barycenter *
27 * 14-nutations, 15-librations) *
28 * center number (same codes as target number) *
29 * coordinate number (1-x, 2-y, ... 6-zdot) *
30 * coordinate [au, au/day]. *
31 * *
32 * For each test case input, testeph *
33 * *
34 * - computes the corresponding state from data contained *
35 * in dexxx, *
36 * *
37 * - compares the two sets, *
38 * *
39 * - writes an error message if the difference between *

41 * *
42 * - writes state and difference information for every npt'th *
43 * test case processed. *

176 ** pleph(et,ntar,ncent,rrd) **
177 ******************************************************************************
178 ** **
179 ** This subroutine reads the jpl planetary ephemeris **
180 ** and gives the position and velocity of the point 'ntarg' **
181 ** with respect to 'ncent'. **
182 ** **
183 ** Calling sequence parameters: **
184 ** **
185 ** et = (double) julian ephemeris date at which interpolation **
186 ** is wanted. **
187 ** **
188 ** ntarg = integer number of 'target' point. **
189 ** **
190 ** ncent = integer number of center point. **
191 ** **
192 ** The numbering convention for 'ntarg' and 'ncent' is: **
193 ** **
194 ** 1 = mercury 8 = neptune **
195 ** 2 = venus 9 = pluto **
196 ** 3 = earth 10 = moon **
197 ** 4 = mars 11 = sun **
198 ** 5 = jupiter 12 = solar-system barycenter **
199 ** 6 = saturn 13 = earth-moon barycenter **
200 ** 7 = uranus 14 = nutations (longitude and obliq) **
201 ** 15 = librations, if on eph. file **
202 ** **
203 ** (If nutations are wanted, set ntarg = 14. **
204 ** For librations, set ntarg = 15. set ncent= 0) **
205 ** **
206 ** rrd = output 6-element, double array of position and velocity **
207 ** of point 'ntarg' relative to 'ncent'. The units are au and **
208 ** au/day. For librations the units are radians and radians **
209 ** per day. In the case of nutations the first four words of **
210 ** rrd will be set to nutations and rates, having units of **
211 ** radians and radians/day. **
212 ** **
213 ** The option is available to have the units in km and km/sec. **
214 ** for this, set km=TRUE at the beginning of the program. **
215 *****************************************************************************/
216 void pleph(double et,int ntarg,int ncent,double rrd[] )
217 {
218  double et2[2],pv[13][6];/* pv is the position/velocity array
219  NUMBERED FROM ZERO: 0=Mercury,1=Venus,...
220  8=Pluto,9=Moon,10=Sun,11=SSBary,12=EMBary
221  First 10 elements (0-9) are affected by state(),
222  all are adjusted here. */
223 
224 
225  int bsave,i,k;
226  int list[12]; /* list is a vector denoting, for which "body"
227  ephemeris values should be calculated by state():
228  0=Mercury,1=Venus,2=EMBary,...,8=Pluto,
229  9=geocentric Moon, 10=nutations in long. & obliq.
230  11= lunar librations */
231 
232 /* initialize et2 for 'state' and set up component count */
233 
234  et2[0]=et;
235  et2[1]=0.0;
236 
237  for(i=0;i<6;++i) rrd[i]=0.0;
238 
239  if(ntarg == ncent) return;
240 
241  for(i=0;i<12;++i) list[i]=0;
242 
243 /* check for nutation call */
244 
245  if(ntarg == 14)
246  {
247  if(R1.r1.ipt[11][1] > 0) /* there is nutation on ephemeris */
248  {
249  list[10]=2;
250  state(et2,list,pv,rrd);
251  }
252  else puts("***** no nutations on the ephemeris file ******\n");
253  return;
254  }
255 
256 /* check for librations */
257 
258  if(ntarg == 15)
259  {
260  if(R1.r1.lpt[1] > 0) /* there are librations on ephemeris file */
261  {
262  list[11]=2;
263  state(et2,list,pv,rrd);
264  for(i=0;i<6;++i) rrd[i]=pv[10][i]; /* librations */
265  }
266  else puts("***** no librations on the ephemeris file *****\n");
267  return;
268  }
269 
270 /* force barycentric output by 'state' */
271 
272  bsave=BARY;
273  BARY= TRUE;
274 
275 /* set up proper entries in 'list' array for state call */
276 
277  for(i=0;i<2;++i) /* list[] IS NUMBERED FROM ZERO ! */
278  {
279  k=ntarg-1;
280  if(i == 1) k=ncent-1; /* same for ntarg & ncent */
281  if(k <= 9) list[k]=2; /* Major planets */
282  if(k == 9) list[2]=2; /* for moon state earth state is necessary*/
283  if(k == 2) list[9]=2; /* for earth state moon state is necessary*/
284  if(k == 12) list[2]=2; /* EMBary state additionally */
285  }
286 
287 /* make call to state */
288 
289  state(et2,list,pv,rrd);
290  /* Solar System barycentric Sun state goes to pv[10][] */
291  if(ntarg == 11 || ncent == 11) for(i=0;i<6;++i) pv[10][i]=PVSUN[i];
292 
293  /* Solar System Barycenter coordinates & velocities equal to zero */
294  if(ntarg == 12 || ncent == 12) for(i=0;i<6;++i) pv[11][i]=0.0;
295 
296  /* Solar System barycentric EMBary state: */
297  if(ntarg == 13 || ncent == 13) for(i=0;i<6;++i) pv[12][i]=pv[2][i];
298 
299  /* if moon from earth or earth from moon ..... */
300  if( (ntarg*ncent) == 30 && (ntarg+ncent) == 13)
301  for(i=0;i<6;++i) pv[2][i]=0.0;
302  else
303  {
304  if(list[2] == 2) /* calculate earth state from EMBary */
305  for(i=0;i<6;++i) pv[2][i] -= pv[9][i]/(1.0+R1.r1.emrat);
306 
307  if(list[9] == 2) /* calculate Solar System barycentric moon state */
308  for(i=0;i<6;++i) pv[9][i] += pv[2][i];
309  }
310 
311  for(i=0;i<6;++i) rrd[i]=pv[ntarg-1][i]-pv[ncent-1][i];
312  BARY=bsave;
313 
314  return;
315 }
316 /*****************************************************************************
317 ** interp(buf,t,ncf,ncm,na,ifl,pv) **
318 ******************************************************************************
319 ** **
320 ** this subroutine differentiates and interpolates a **
321 ** set of chebyshev coefficients to give position and velocity **
322 ** **
323 ** calling sequence parameters: **
324 ** **
325 ** input: **
326 ** **
327 ** buf 1st location of array of d.p. chebyshev coefficients **
328 ** of position **
329 ** **
330 ** t t[0] is double fractional time in interval covered by **
331 ** coefficients at which interpolation is wanted **
332 ** (0 <= t[0] <= 1). t[1] is dp length of whole **
333 ** interval in input time units. **
334 ** **
335 ** ncf # of coefficients per component **
336 ** **
337 ** ncm # of components per set of coefficients **
338 ** **
339 ** na # of sets of coefficients in full array **
340 ** (i.e., # of sub-intervals in full interval) **
341 ** **
342 ** ifl integer flag: =1 for positions only **
343 ** =2 for pos and vel **
344 ** **
345 ** **
346 ** output: **
347 ** **
348 ** pv interpolated quantities requested. dimension **
349 ** expected is pv(ncm,ifl), dp. **
350 ** **
351 *****************************************************************************/
352 void interp(double coef[],double t[2],int ncf,int ncm,int na,int ifl,
353  double posvel[6])
354 {
355  static double pc[18],vc[18];
356  static int np=2, nv=3, first=1;
357  static double twot=0.0;
358  double dna,dt1,temp,tc,vfac,temp1;
359  int l,i,j;
360 
361  if(first){ /* initialize static vectors when called first time */
362  pc[0]=1.0;
363  pc[1]=0.0;
364  vc[1]=1.0;
365  first=0;
366  }
367 
368 /* entry point. get correct sub-interval number for this set
369  of coefficients and then get normalized chebyshev time
370  within that subinterval. */
371 
372  dna=(double)na;
373  modf(t[0],&dt1);
374  temp=dna*t[0];
375  l=(int)(temp-dt1);
376 
377 /* tc is the normalized chebyshev time (-1 <= tc <= 1) */
378 
379  tc=2.0*(modf(temp,&temp1)+dt1)-1.0;
380 
381 /* check to see whether chebyshev time has changed,
382  and compute new polynomial values if it has.
383  (the element pc[1] is the value of t1[tc] and hence
384  contains the value of tc on the previous call.) */
385 
386  if(tc != pc[1])
387  {
388  np=2;
389  nv=3;
390  pc[1]=tc;
391  twot=tc+tc;
392  }
393 
394 /* be sure that at least 'ncf' polynomials have been evaluated
395  and are stored in the array 'pc'. */
396 
397  if(np < ncf)
398  {
399  for(i=np;i<ncf;++i) pc[i]=twot*pc[i-1]-pc[i-2];
400  np=ncf;
401  }
402 
403 /* interpolate to get position for each component */
404 
405  for(i=0;i<ncm;++i) /* ncm is a number of coordinates */
406  {
407  posvel[i]=0.0;
408  for(j=ncf-1;j>=0;--j)
409  posvel[i]=posvel[i]+pc[j]*coef[j+i*ncf+l*ncf*ncm];
410  }
411 
412  if(ifl <= 1) return;
413 
414 
415 /* if velocity interpolation is wanted, be sure enough
416  derivative polynomials have been generated and stored. */
417 
418  vfac=(dna+dna)/t[1];
419  vc[2]=twot+twot;
420  if(nv < ncf)
421  {
422  for(i=nv;i<ncf;++i) vc[i]=twot*vc[i-1]+pc[i-1]+pc[i-1]-vc[i-2];
423  nv=ncf;
424  }
425 
426 /* interpolate to get velocity for each component */
427 
428  for(i=0;i<ncm;++i)
429  {
430  posvel[i+ncm]=0.0;
431  for(j=ncf-1;j>0;--j)
432  posvel[i+ncm]=posvel[i+ncm]+vc[j]*coef[j+i*ncf+l*ncf*ncm];
433  posvel[i+ncm]=posvel[i+ncm]*vfac;
434  }
435  return;
436 }

471 /*****************************************************************************
472 ** state(et2,list,pv,nut) **
473 ******************************************************************************
474 ** This subroutine reads and interpolates the jpl planetary ephemeris file **
475 ** **
476 ** Calling sequence parameters: **
477 ** **
478 ** Input: **
479 ** **
480 ** et2[] double, 2-element JED epoch at which interpolation **
481 ** is wanted. Any combination of et2[0]+et2[1] which falls **
482 ** within the time span on the file is a permissible epoch. **
483 ** **
484 ** a. for ease in programming, the user may put the **
485 ** entire epoch in et2[0] and set et2[1]=0.0 **
486 ** **
487 ** b. for maximum interpolation accuracy, set et2[0] = **
488 ** the most recent midnight at or before interpolation **
489 ** epoch and set et2[1] = fractional part of a day **
490 ** elapsed between et2[0] and epoch. **
491 ** **
492 ** c. as an alternative, it may prove convenient to set **
493 ** et2[0] = some fixed epoch, such as start of integration,**
494 ** and et2[1] = elapsed interval between then and epoch. **
495 ** **
496 ** list 12-element integer array specifying what interpolation **
497 ** is wanted for each of the "bodies" on the file. **
498 ** **
499 ** list[i]=0, no interpolation for body i **
500 ** =1, position only **
501 ** =2, position and velocity **
502 ** **
503 ** the designation of the astronomical bodies by i is: **
504 ** **
505 ** i = 0: mercury **
506 ** = 1: venus **
507 ** = 2: earth-moon barycenter **
508 ** = 3: mars **
509 ** = 4: jupiter **
510 ** = 5: saturn **
511 ** = 6: uranus **
512 ** = 7: neptune **
513 ** = 8: pluto **
514 ** = 9: geocentric moon **
515 ** =10: nutations in longitude and obliquity **
516 ** =11: lunar librations (if on file) **
517 ** **
518 ** **
519 ** output: **
520 ** **
521 ** pv[][6] double array that will contain requested interpolated **
522 ** quantities. The body specified by list[i] will have its **
523 ** state in the array starting at pv[i][0] (on any given **
524 ** call, only those words in 'pv' which are affected by the **
525 ** first 10 'list' entries (and by list(11) if librations are **
526 ** on the file) are set. The rest of the 'pv' array **
527 ** is untouched.) The order of components in pv[][] is: **
528 ** pv[][0]=x,....pv[][5]=dz. **
529 ** **
530 ** All output vectors are referenced to the earth mean **
531 ** equator and equinox of epoch. The moon state is always **
532 ** geocentric; the other nine states are either heliocentric **
533 ** or solar-system barycentric, depending on the setting of **
534 ** global variables (see below). **
535 ** **
536 ** Lunar librations, if on file, are put into pv[10][k] if **
537 ** list[11] is 1 or 2. **
538 ** **
539 ** nut dp 4-word array that will contain nutations and rates, **
540 ** depending on the setting of list[10]. the order of **
541 ** quantities in nut is: **
542 ** **
543 ** d psi (nutation in longitude) **
544 ** d epsilon (nutation in obliquity) **
545 ** d psi dot **
546 ** d epsilon dot **
547 ** **
548 ** Global variables: **
549 ** **
550 ** KM logical flag defining physical units of the output **
551 ** states. KM = TRUE, km and km/sec **
552 ** = FALSE, au and au/day **
553 ** default value = FALSE. (Note that KM determines time unit **
554 ** for nutations and librations. Angle unit is always radians.)**
555 ** **
556 ** BARY logical flag defining output center. **
557 ** only the 9 planets (entries 0-8) are affected. **
558 ** BARY = TRUE =\ center is solar-system barycenter **
559 ** = FALSE =\ center is sun **
560 ** default value = FALSE **
561 ** **
562 ** PVSUN double, 6-element array containing **
563 ** the barycentric position and velocity of the sun. **
564 ** **
565 *****************************************************************************/
566 void state(double et2[2],int list[12],double pv[][6],double nut[4])
567 {
568  int i,j;
569  static int ipt[13][3], first=TRUE;
570  /* local copy of R1.r1.ipt[] is extended for R1.r1.lpt[[] */
571  long int nr;
572  static long int nrl=0;
573  double pjd[4];
574  static double buf[NCOEFF];
575  double s,t[2],aufac;
576  double pefau[6];
577 
578  if(first)
579  {
580  first=FALSE;
581  for(i=0;i<3;++i)
582  {
583  for(j=0;j<12;++j) ipt[j][i]=(int)R1.r1.ipt[j][i];
584  ipt[12][i] = (int)R1.r1.lpt[i];
585  }
586  }
587 
588 /* ********** main entry point ********** */
589 
590  s=et2[0] - 0.5;
591  split(s,&pjd[0]);
592  split(et2[1],&pjd[2]);
593  pjd[0]=pjd[0]+pjd[2]+0.5;
594  pjd[1]=pjd[1]+pjd[3];
595  split(pjd[1],&pjd[2]);
596  pjd[0]=pjd[0]+pjd[2];
597 /* here pjd[0] contains last midnight before epoch desired (in JED: *.5)
598  and pjd[3] contains the remaining, fractional part of the epoch */
599 
600 /* error return for epoch out of range */
601 
602  if( (pjd[0]+pjd[3]) < R1.r1.ss[0] || (pjd[0]+pjd[3]) > R1.r1.ss[1] )
603  {
604  puts("Requested JED not within ephemeris limits.\n");
605  return;
606  }
607 
608 /* calculate record # and relative time in interval */
609 
610  nr=(long)((pjd[0]-R1.r1.ss[0])/R1.r1.ss[2])+2;
611  /* add 2 to adjust for the first two records containing header data */
612  if(pjd[0] == R1.r1.ss[1]) nr=nr-1;
613  t[0]=( pjd[0]-( (1.0*nr-2.0)*R1.r1.ss[2]+R1.r1.ss[0] ) +
614  pjd[3] )/R1.r1.ss[2];
615
